Create `src/effects/Particle.ts` — a poolable particle entity:

```typescript
import { Mesh, BufferGeometry, MeshStandardMaterial, Color, Float32BufferAttribute } from 'three';
import type { Scene } from 'three';

/** A single pooled particle — geometric shard or flash dot. */
export class Particle {
  public active: boolean = false;
  public readonly mesh: Mesh;

  // Physics state
  public x: number = 0;
  public y: number = 0;
  public vx: number = 0;   // units/second
  public vy: number = 0;
  public rotationSpeed: number = 0; // radians/second (tumble)
  public lifetime: number = 0;      // remaining seconds
  public maxLifetime: number = 0;

  constructor(scene: Scene) {
    // Diamond shard geometry — 4 vertices forming a flattened diamond
    // Width=6, Height=10 — angular shard shape (Claude's discretion)
    const geo = new BufferGeometry();
    geo.setAttribute('position', new Float32BufferAttribute([
       0,  5, 0,  // top
      -3,  0, 0,  // left
       0, -5, 0,  // bottom
       3,  0, 0,  // right
    ], 3));
    geo.setIndex([0, 1, 2, 0, 2, 3]);  // two triangles
    const mat = new MeshStandardMaterial({
      color: 0xffffff,
      emissive: new Color(0xffffff),
      emissiveIntensity: 1.2,
      roughness: 1.0,
      metalness: 0.0,
    });
    this.mesh = new Mesh(geo, mat);
    this.mesh.visible = false;
    scene.add(this.mesh);
  }

  /** Initialize this particle for a death shard burst. */
  public init(
    x: number, y: number,
    vx: number, vy: number,
    rotationSpeed: number,
    lifetime: number,
    hexColor: number,
  ): void {
    this.active = true;
    this.x = x; this.y = y;
    this.vx = vx; this.vy = vy;
    this.rotationSpeed = rotationSpeed;
    this.lifetime = lifetime;
    this.maxLifetime = lifetime;
    const mat = this.mesh.material as MeshStandardMaterial;
    mat.color.setHex(hexColor);
    mat.emissive.setHex(hexColor);
    this.mesh.position.set(x, y, 0.5);
    this.mesh.rotation.z = Math.random() * Math.PI * 2;
    this.mesh.visible = true;
  }

  /** Update particle physics. Returns true if still alive. */
  public update(dt: number): boolean {
    if (!this.active) return false;
    this.lifetime -= dt;
    if (this.lifetime <= 0) {
      this.active = false;
      this.mesh.visible = false;
      return false;
    }
    // Move
    this.x += this.vx * dt;
    this.y += this.vy * dt;
    // Tumble
    this.mesh.rotation.z += this.rotationSpeed * dt;
    // Fade out: scale emissiveIntensity by remaining lifetime ratio
    const t = this.lifetime / this.maxLifetime;
    const mat = this.mesh.material as MeshStandardMaterial;
    mat.emissiveIntensity = 1.2 * t;
    // Scale down as they die (shrink to zero)
    const s = 0.5 + 0.5 * t;
    this.mesh.scale.setScalar(s);
    this.mesh.position.set(this.x, this.y, 0.5);
    return true;
  }

  /** Release back to pool */
  public reset(): void {
    this.active = false;
    this.mesh.visible = false;
  }
}
```

Create `src/effects/ParticleManager.ts` — manages three particle pools:

```typescript
import { ObjectPool } from '../core/ObjectPool';
import { Particle } from './Particle';
import { wavePalette } from '../config/palettes';
import { runState } from '../state/RunState';
import type { Scene } from 'three';

// Pool sizes: death bursts (6-8 per kill, up to 40 enemies)
const DEATH_PARTICLE_POOL_SIZE = 128;
// Muzzle flash: brief, few particles, fast recycle
const MUZZLE_PARTICLE_POOL_SIZE = 16;
// Engine trail: continuous but short-lived
const TRAIL_PARTICLE_POOL_SIZE = 32;

export class ParticleManager {
  private readonly deathPool: ObjectPool<Particle>;
  private readonly muzzlePool: ObjectPool<Particle>;
  private readonly trailPool: ObjectPool<Particle>;

  private readonly activeParticles: Particle[] = [];

  constructor(scene: Scene) {
    this.deathPool = new ObjectPool(() => new Particle(scene), DEATH_PARTICLE_POOL_SIZE);
    this.muzzlePool = new ObjectPool(() => new Particle(scene), MUZZLE_PARTICLE_POOL_SIZE);
    this.trailPool  = new ObjectPool(() => new Particle(scene), TRAIL_PARTICLE_POOL_SIZE);
  }

  /**
   * Spawn 6-8 geometric shard particles at enemy kill position.
   * Color matches the current wave palette.
   */
  public spawnDeathBurst(x: number, y: number, hexColor: number): void {
    const count = 6 + Math.floor(Math.random() * 3); // 6-8 shards
    for (let i = 0; i < count; i++) {
      const p = this.deathPool.acquire();
      if (!p) break;
      // Tight burst: radial velocity 80-160 units/second (not wide scatter)
      const angle = (i / count) * Math.PI * 2 + (Math.random() - 0.5) * 0.5;
      const speed = 80 + Math.random() * 80;
      const vx = Math.cos(angle) * speed;
      const vy = Math.sin(angle) * speed;
      const rotSpeed = (Math.random() - 0.5) * 8; // ±4 rad/sec tumble
      const lifetime = 0.4 + Math.random() * 0.25; // 0.4-0.65 seconds
      p.init(x, y, vx, vy, rotSpeed, lifetime, hexColor);
      this.activeParticles.push(p);
    }
  }

  /**
   * Spawn a brief muzzle flash at the bullet spawn point.
   * 3-4 white particles, fast scatter, short lifetime.
   */
  public spawnMuzzleFlash(x: number, y: number): void {
    const count = 3 + Math.floor(Math.random() * 2); // 3-4 flash particles
    for (let i = 0; i < count; i++) {
      const p = this.muzzlePool.acquire();
      if (!p) break;
      const angle = (Math.random() - 0.5) * Math.PI; // spread upward ~180° arc
      const speed = 40 + Math.random() * 60;
      const vx = Math.cos(angle + Math.PI / 2) * speed * 0.3;
      const vy = Math.sin(angle + Math.PI / 2) * speed;
      const rotSpeed = (Math.random() - 0.5) * 12;
      const lifetime = 0.1 + Math.random() * 0.1; // very brief: 0.1-0.2 seconds
      p.init(x, y, vx, vy, rotSpeed, lifetime, 0xffffff); // always white flash
      this.activeParticles.push(p);
    }
  }

  /**
   * Spawn a single engine trail particle behind the player ship.
   * Called each fixed step while player is moving horizontally.
   * x, y = player position; movingRight = direction for offset
   */
  public spawnEngineTrail(x: number, y: number): void {
    const p = this.trailPool.acquire();
    if (!p) return;
    // Trail spawns just below the player ship (at the engine exhaust position)
    const trailX = x + (Math.random() - 0.5) * 8;
    const trailY = y - 12; // below the ship base
    const vx = (Math.random() - 0.5) * 20;
    const vy = -(20 + Math.random() * 30); // drift downward
    const rotSpeed = (Math.random() - 0.5) * 6;
    const lifetime = 0.2 + Math.random() * 0.15;
    const color = wavePalette.getColor(runState.wave);
    p.init(trailX, trailY, vx, vy, rotSpeed, lifetime, color);
    this.activeParticles.push(p);
  }

  /** Update all active particles. Call each fixed step. */
  public update(dt: number): void {
    for (let i = this.activeParticles.length - 1; i >= 0; i--) {
      const p = this.activeParticles[i];
      const alive = p.update(dt);
      if (!alive) {
        // Determine which pool to release to by checking pool membership
        // Simple approach: release to death pool by default (all particles same class)
        // Since pools share the same Particle class, we can track by pool
        // Better: store pool reference on particle or use a combined release method
        this.releaseParticle(p);
        this.activeParticles.splice(i, 1);
      }
    }
  }

  private releaseParticle(p: Particle): void {
    // Try each pool — release() is safe to call on items not in the pool
    // (ObjectPool.release sets active=false and returns to pool; calling on wrong pool is benign if pool has space)
    // Track pool by tagging: store a 'poolType' on particle, or use a single shared pool
    // Simplest: use a SINGLE combined pool for all particle types, sized to the sum
    // REVISION: use a single allParticlesPool approach OR track which pool acquired each particle
    // RECOMMENDED: Refactor to single pool (sum of sizes) to simplify release logic:
    // private readonly particlePool: ObjectPool<Particle> (size = 128+16+32 = 176)
    // Then release is always to the same pool. Adjust constructor accordingly.
    p.reset();
    this.deathPool.release(p);  // This works only if particle came from deathPool
    // ACTUAL IMPLEMENTATION: Use a Map<Particle, ObjectPool<Particle>> to track source pool
  }

  /**
   * Register all pooled particle meshes with the bloom selection.
   * Called once from Game.init() after initBloom().
   */
  public registerBloom(addToBloom: (mesh: import('three').Object3D) => void): void {
    this.deathPool.forEach(p => addToBloom(p.mesh));
    this.muzzlePool.forEach(p => addToBloom(p.mesh));
    this.trailPool.forEach(p => addToBloom(p.mesh));
  }
}
```
