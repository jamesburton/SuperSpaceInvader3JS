  <action>
    Create src/entities/PickupToken.ts.

    PickupToken is a floating pickup collectible that drifts downward slowly after spawning. The player ship collides with it to collect the power-up (collision handled by CollisionSystem in Plan 03-06).

    ```typescript
    import { Mesh, BoxGeometry, MeshStandardMaterial } from 'three';
    import type { Scene } from 'three';
    import type { PowerUpType } from '../config/powerups';
    import { POWER_UP_DEFS } from '../config/powerups';

    const TOKEN_SIZE = 14;        // square bounding box (half = 7)
    const TOKEN_DRIFT_SPEED = 40; // units/second downward drift

    export class PickupToken {
      public x: number = 0;
      public y: number = 0;
      public vy: number = -TOKEN_DRIFT_SPEED;
      public readonly width: number = 7;   // AABB half-width
      public readonly height: number = 7;  // AABB half-height
      public active: boolean = false;
      public type: PowerUpType = 'spreadShot';
      declare public visible: boolean;
      public readonly mesh: Mesh;

      constructor(scene: Scene) {
        const geo = new BoxGeometry(TOKEN_SIZE, TOKEN_SIZE, 1);
        const mat = new MeshStandardMaterial({
          color: 0x00ffff,
          emissive: 0x00ffff,
          emissiveIntensity: 1.5,
          roughness: 1.0,
          metalness: 0.0,
        });
        this.mesh = new Mesh(geo, mat);
        this.mesh.visible = false;
        scene.add(this.mesh);
      }

      public init(x: number, y: number, type: PowerUpType): void {
        this.x = x; this.y = y;
        this.type = type;
        this.active = true;
        this.vy = -TOKEN_DRIFT_SPEED;
        this.mesh.position.set(x, y, 0);
        const mat = this.mesh.material as MeshStandardMaterial;
        const def = POWER_UP_DEFS[type];
        mat.color.setHex(def.color);
        mat.emissive.setHex(def.color);
        // Rotate 45 degrees for a diamond appearance
        this.mesh.rotation.z = Math.PI / 4;
      }

      /** Move token downward. Returns false when token exits world bottom. */
      public update(dt: number): boolean {
        this.y += this.vy * dt;
        this.mesh.position.y = this.y;
        // Expire when below world bottom (-WORLD_HEIGHT/2 - 50 gives headroom)
        return this.y > -350; // WORLD_HEIGHT/2 = 300, +50 buffer
      }
    }
    ```

  <name>Task 3: Create PowerUpManager</name>
  <files>src/systems/PowerUpManager.ts</files>
  <action>
    Create src/systems/PowerUpManager.ts.

    PowerUpManager owns:
    1. A pool of 16 PickupTokens (enemy kills are infrequent; 16 tokens is generous)
    2. Active power-up state: which type is active and how much time remains
    3. Spawn method: called by CollisionSystem after enemy kill (Plan 03-06)
    4. Collect method: called by CollisionSystem when player touches token
    5. Update: ticks active power-up duration and drifts tokens downward

    ```typescript
    import type { Scene } from 'three';
    import { ObjectPool } from '../core/ObjectPool';
    import { PickupToken } from '../entities/PickupToken';
    import type { PowerUpType } from '../config/powerups';
    import { POWER_UP_DEFS, POWER_UP_TYPES } from '../config/powerups';
    import { WORLD_HEIGHT } from '../utils/constants';

    const PICKUP_POOL_SIZE = 16;

    export class PowerUpManager {
      private readonly pool: ObjectPool<PickupToken>;
      private readonly activeTokens: PickupToken[] = [];

      // Active power-up state (only one power-up active at a time — last collected wins)
      private activePowerUp: PowerUpType | null = null;
      private activeDuration: number = 0;

      // Shield absorb state (Shield power-up: absorbs one hit, then expires)
      private shieldCharges: number = 0;

      constructor(scene: Scene) {
        this.pool = new ObjectPool<PickupToken>(
          () => new PickupToken(scene),
          PICKUP_POOL_SIZE,
        );
      }

      /**
       * Try to spawn a power-up drop at (x, y) using the given dropChance.
       * Selects a random PowerUpType uniformly.
       * Called by CollisionSystem after enemy kill.
       */
      public trySpawnDrop(x: number, y: number, dropChance: number): void {
        if (Math.random() > dropChance) return;
        const token = this.pool.acquire();
        if (!token) return;
        const type = POWER_UP_TYPES[Math.floor(Math.random() * POWER_UP_TYPES.length)];
        token.init(x, y, type);
        this.activeTokens.push(token);
      }

      /**
       * Activate a collected power-up. Last collected wins (replaces current).
       * Shield type: adds a shield charge instead of a duration.
       * Returns the display name for PickupFeedback.
       */
      public collectPickup(token: PickupToken): string {
        const def = POWER_UP_DEFS[token.type];
        if (token.type === 'shield') {
          this.shieldCharges = 1; // one absorb charge
        } else {
          this.activePowerUp = token.type;
          this.activeDuration = def.duration;
        }
        this.pool.release(token);
        const idx = this.activeTokens.indexOf(token);
        if (idx !== -1) this.activeTokens.splice(idx, 1);
        return def.displayName;
      }

      /** Query whether a power-up type is currently active */
      public isActive(type: PowerUpType): boolean {
        if (type === 'shield') return this.shieldCharges > 0;
        return this.activePowerUp === type && this.activeDuration > 0;
      }

      /**
       * Called when player takes a hit while shield is active.
       * Consumes one shield charge and returns true (hit absorbed).
       */
      public consumeShieldCharge(): boolean {
        if (this.shieldCharges > 0) {
          this.shieldCharges = 0;
          this.activePowerUp = null; // clear any concurrent active
          return true; // absorbed
        }
        return false;
      }

      /** Get remaining duration for current active power-up (0 if none). For HUD display. */
      public get activeDurationRemaining(): number {
        return this.activeDuration;
      }

      /** Get full duration for active power-up type (for HUD percentage bar). */
      public get activeDurationFull(): number {
        if (!this.activePowerUp) return 0;
        return POWER_UP_DEFS[this.activePowerUp].duration;
      }

      public get activePowerUpType(): PowerUpType | null {
        return this.activePowerUp;
      }

      public get activeShieldCharges(): number {
        return this.shieldCharges;
      }

      /**
       * Update all active tokens (drift down) and tick active power-up duration.
       * Call each fixed step.
       */
      public update(dt: number): void {
        // Tick active power-up
        if (this.activePowerUp && this.activeDuration > 0) {
          this.activeDuration -= dt;
          if (this.activeDuration <= 0) {
            this.activeDuration = 0;
            this.activePowerUp = null;
          }
        }

        // Move tokens + cull off-screen ones
        for (let i = this.activeTokens.length - 1; i >= 0; i--) {
          const token = this.activeTokens[i];
          const alive = token.update(dt);
          if (!alive) {
            this.pool.release(token);
            this.activeTokens.splice(i, 1);
          }
        }
      }

      /** Release all tokens — call on wave reset or run end */
      public releaseAll(): void {
        for (const token of this.activeTokens) {
          this.pool.release(token);
        }
        this.activeTokens.length = 0;
        this.activePowerUp = null;
        this.activeDuration = 0;
        this.shieldCharges = 0;
      }

      /** Get active tokens array — for CollisionSystem to iterate player-vs-token AABBs */
      public getActiveTokens(): readonly PickupToken[] {
        return this.activeTokens;
      }

      /** Register all token meshes with bloom selection. Call once from Game.init(). */
      public registerBloom(addToBloom: (mesh: import('three').Object3D) => void): void {
        this.pool.forEach(t => addToBloom(t.mesh));
      }
    }
    ```
