    ```typescript
    private fireShot(player, playerBulletPool, activeBullets, powerUpManager?): void {
      const isSpread = powerUpManager?.isActive('spreadShot') ?? false;
      const angles = isSpread ? [-0.25, 0, 0.25] : [0]; // radians offset

      for (const angleOffset of angles) {
        const bullet = playerBulletPool.acquire();
        if (!bullet) break;
        bullet.init(player.x, player.y + player.height + 10, true);
        if (isSpread) {
          // Decompose vy into angled trajectory
          const speed = Math.abs(bullet.vy); // BULLET_SPEED
          bullet.vx = Math.sin(angleOffset) * speed;
          bullet.vy = Math.cos(angleOffset) * speed;
          bullet.setColor(0x0088ff); // spread = blue (FEEL-07)
        }
        activeBullets.push(bullet);
      }
    }
    ```
