**Apply meta bonuses in PlayingState.enter():**

In PlayingState, add a private method `applyMetaBonuses()` and call it from `enter()`:

```typescript
enter(): void {
  this.lastWasTransitioning = false;
  runState.setPhase('playing');
  this.hud.hideOverlay();
  this.ctx.cameraShake.reset();
  this.applyMetaBonuses(); // Phase 4: apply persistent meta upgrades
}

private applyMetaBonuses(): void {
  const { purchasedUpgrades } = useMetaStore.getState();
  const { player, powerUpManager, shopSystem } = this.ctx;

  // Count tiers for multiplicative passives
  let fireRateTiers = 0;
  let moveSpeedTiers = 0;

  for (const id of purchasedUpgrades) {
    if (id.startsWith('passive_fireRate_')) fireRateTiers++;
    if (id.startsWith('passive_moveSpeed_')) moveSpeedTiers++;
  }

  // Apply passive fire rate: +10% per tier (compound)
  if (fireRateTiers > 0) {
    const multiplier = Math.pow(1.10, fireRateTiers); // e.g., 3 tiers = 1.331x
    // Decrease cooldown by setting multiplier < 1 (faster = shorter cooldown)
    player.setFireCooldownMultiplier(1 / multiplier);
  }

  // Apply passive move speed: +8% per tier (compound)
  if (moveSpeedTiers > 0) {
    const multiplier = Math.pow(1.08, moveSpeedTiers);
    player.setSpeedMultiplier(multiplier);
  }

  // Apply starting life bonus
  if (purchasedUpgrades.includes('passive_startingLife')) {
    runState.addLife(); // adds 1 life capped at MAX_LIVES_CAP
  }

  // Apply starting loadout power-ups (timed, 30s duration)
  if (purchasedUpgrades.includes('loadout_spread_start')) {
    // Activate spread shot for 30 seconds at run start
    powerUpManager.activate('spread', 30);
  }
  if (purchasedUpgrades.includes('loadout_rapid_start')) {
    // Activate rapid fire for 30 seconds at run start
    powerUpManager.activate('rapid', 30);
  }
}
```
