    - `makeChargerGeometry()` — Heavy rhombus/wedge with a blunt nose pointing down (charge direction). 6 vertices. Wider than tall, nose points downward. Color identity: hot orange (0xff6600).
    - `makeSwooperGeometry()` — Small arc/crescent shape, concave side facing forward, suggesting sweep motion. 8 vertices forming a curved wing. Color identity: lime green (0x00ff44).

    class GridFormationLayout implements FormationLayout {
      constructor(public readonly rows: number, public readonly cols: number,
                  public readonly colSpacing: number, public readonly rowSpacing: number) {}
      getPosition(row, col, formationX, formationY) {
        return {
          x: formationX + (col - (this.cols - 1) / 2) * this.colSpacing,
          y: formationY - row * this.rowSpacing,
        };
      }
    }
