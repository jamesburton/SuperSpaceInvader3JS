```typescript
export class BossHealthBar {
  private readonly root: HTMLElement;
  private readonly fillEl: HTMLElement;
  private readonly phaseMarkerEl: HTMLElement;
  private readonly phaseLabel: HTMLElement;

  constructor(hudRoot: HTMLElement) {
    const el = document.createElement('div');
    el.id = 'boss-health-bar';
    el.style.cssText = [
      'display:none',
      'position:absolute',
      'bottom:32px',
      'left:50%',
      'transform:translateX(-50%)',
      'width:60%',
      'background:transparent',
      'text-align:center',
      'font-family:"Courier New",monospace',
    ].join(';');

    el.innerHTML = `
      <div id="boss-phase-label" style="font-size:12px;color:#ff1133;text-shadow:0 0 8px #ff1133;margin-bottom:4px;letter-spacing:2px;">BOSS — PHASE 1</div>
      <div style="position:relative;width:100%;height:12px;background:#111;border:1px solid #ff1133;box-shadow:0 0 8px #ff1133;">
        <div id="boss-hp-fill" style="height:100%;width:100%;background:#ff1133;transition:width 0.1s linear;"></div>
        <div id="boss-phase-marker" style="position:absolute;top:0;left:50%;width:2px;height:100%;background:#fff;opacity:0.7;"></div>
      </div>
    `;

    hudRoot.appendChild(el);
    this.root = el;
    this.fillEl = el.querySelector('#boss-hp-fill') as HTMLElement;
    this.phaseMarkerEl = el.querySelector('#boss-phase-marker') as HTMLElement;
    this.phaseLabel = el.querySelector('#boss-phase-label') as HTMLElement;
  }

  public show(_totalPhases: number, _currentPhase: number): void {
    this.root.style.display = 'block';
    this.update(1.0, 1);
  }

  public hide(): void {
    this.root.style.display = 'none';
  }

  public update(healthPercent: number, currentPhase: number): void {
    const pct = Math.max(0, Math.min(1, healthPercent)) * 100;
    const color = currentPhase >= 2 ? '#FF6600' : '#FF1133';
    const glow = currentPhase >= 2 ? '#FF6600' : '#FF1133';
    this.fillEl.style.width = `${pct}%`;
    this.fillEl.style.background = color;
    this.root.style.borderColor = glow;
    this.root.style.boxShadow = `0 0 8px ${glow}`;
    this.phaseLabel.style.color = color;
    this.phaseLabel.style.textShadow = `0 0 8px ${glow}`;
    this.phaseLabel.textContent = `BOSS — PHASE ${currentPhase}`;
    // Phase marker always at 50% (boundary between phase 1 and 2)
    this.phaseMarkerEl.style.display = 'block';
  }
}
```

Add a private method `updateBossCollision(dt: number)` to PlayingState that handles player bullets hitting boss AABB:

```typescript
private updateBossCollision(dt: number): void {
  // Note: dt unused directly, included for signature consistency
  const { ctx } = this;
  const boss = ctx.boss;
  if (!boss.active) return;
  const toRelease: typeof ctx.activeBullets[number][] = [];
  for (const bullet of ctx.activeBullets) {
    if (!bullet.active || !bullet.isPlayerBullet) continue;
    const dx = Math.abs(bullet.x - boss.x);
    const dy = Math.abs(bullet.y - boss.y);
    if (dx < boss.width + 2 && dy < boss.height + 2) {
      boss.takeDamage(1);
      runState.addScore(5);  // small score per hit (not per-kill)
      bullet.active = false;
      toRelease.push(bullet);
      // Death burst on boss at hit location
      ctx.particleManager.spawnDeathBurst(bullet.x, bullet.y, 0xFF1133);
    }
  }
  // Remove hit player bullets
  for (const b of toRelease) {
    ctx.playerBulletPool.release(b);
  }
  ctx.activeBullets.splice(0, ctx.activeBullets.length,
    ...ctx.activeBullets.filter(b => b.active));
}
```

Note: The normal `ctx.activeBullets` cleanup for released bullets may already be handled by MovementSystem culling — check the pattern in MovementSystem.updateBullets. If it filters by `b.active`, set `bullet.active = false` and let the existing filter handle it in the next iteration. Adapt accordingly to avoid double-mutation.
